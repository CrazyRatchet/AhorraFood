// Reglas de Firebase Storage recomendadas para AhorraFood
// Copia y pega esto en Firebase Console → Storage → Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Permitir que los comercios suban imágenes de productos en la carpeta FotosProducto
    match /FotosProducto/{userId}/{allPaths=**} {
      allow read: if true;  // Cualquiera puede ver las fotos de productos
      allow write: if request.auth != null && request.auth.uid == userId;  // Solo el dueño puede subir
    }
    
    // Carpeta de test para debugging
    match /FotosProducto/test/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
